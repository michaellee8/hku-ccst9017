\begin{question}{Q. 2}
	Use Fermatâ€™s little theorem to find the remainder when $2^{123}$ is divided by $29$. 
\end{question}
\begin{answer}{Q. 2}
	\hfill \break
    Since $29$ is a prime number, we have
    $$ 2^{28} \equiv 1  \mod {29}$$
    hence we have
    \begin{align}
        2^{123} &\equiv 2^{28\times 4 + 11} \mod 29 \\
        2^{123} &\equiv (2^{28})^4  (2^{11}) \mod 29 \\
        2^{123} &\equiv  2^{11} \mod 29 \\
        2^{123} &\equiv  2048 \mod 29 \\
        2^{123} &\equiv  18 \mod 29 
    \end{align}
\end{answer}
